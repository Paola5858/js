// setTimeout sem promise — js não espera

assíncrono sem promise

O JS não espera o setTimeout

O setTimeout agenda um código pra rodar depois, mas o JS não para — ele segue pra próxima linha imediatamente. A ordem que aparece no console não é a ordem do código.

console.log("1. Mandei o currículo");

setTimeout(() => {
    // isso roda 3 segundos depois
    console.log("2. A empresa respondeu: VOCÊ FOI CONTRATADO!");
}, 3000);

// JS não espera o setTimeout acima
console.log("3. Continuei estudando outras coisas...");
console ao vivo
// clique em executar para ver a ordem
0ms linha 1 executa imediatamente
0ms setTimeout é agendado, JS segue em frente
0ms linha 3 executa imediatamente
3s callback do setTimeout finalmente roda

// promise com resolve + async/await — funciona direitinho

async/await com resolve

fazerCafe() + rotinaMatinal()

A Promise envolve o setTimeout e o resolve é chamado quando o tempo acaba. O await pausa a função até a promise resolver, depois continua de onde parou.

function fazerCafe() {
    return new Promise((resolve) => {
        setTimeout(() => {
            resolve("Café pronto e quentinho!");
        }, 2000);
    });
}

async function rotinaMatinal() {
    console.log("Acordei e estou com sono...");

    // await pausa aqui até a promise resolver
    const cafe = await fazerCafe();

    console.log(cafe);
    console.log("Agora sim, partiu trabalho!");
}

rotinaMatinal();
console ao vivo
// clique em executar para ver o await pausar
0ms "Acordei..." imprime na hora
2s await pausa até o resolve ser chamado
2s "Café pronto..." e "partiu trabalho!" imprimem

// promise sem resolve — travada para sempre

promise travada — bug de propósito

escolha() — o resolve sumiu

O setTimeout roda e o log aparece, mas o resolve nunca é chamado. A promise fica em estado pending pra sempre. O await fica esperando eternamente — o código depois dele nunca executa.

function escolha() {
    return new Promise((resolve) => {
        setTimeout(() => {
            console.log("Cliente escolheu a pizza de calabresa!");
            // faltou chamar: resolve("pizza de calabresa")
            // sem resolve → promise fica "pending" pra sempre
        }, 3000);
    });
}

async function assarPizza() {
    console.log("Iniciando o processo de assar a pizza...");
    console.log("Esperando o cliente escolher a pizza...");

    const pizza = await escolha(); // trava aqui pra sempre

    console.log(pizza); // nunca executa
}

assarPizza();
console ao vivo
// clique em executar para ver a promise travar
0ms "Iniciando..." e "Esperando..." imprimem
3s setTimeout roda, o log da pizza aparece
∞ await nunca avança — resolve nunca foi chamado
para corrigir: adicione resolve("pizza de calabresa") dentro do setTimeout, logo após o console.log. sem isso, o console.log(pizza) nunca vai rodar.

// fetch + async/await — consumindo uma api real

fetch · async/await · API Xano

buscarMarcas() — dados da nuvem

O fetch retorna uma promise assim como o fazerCafe. Precisa de dois await: um pra esperar a resposta da rede, outro pra converter o body em JSON.

async function buscarMarcas() {
    const url = 'https://x8ki-letl-twmt.n7.xano.io/api:ijUECDHD/marca';

    // await 1: espera a rede responder
    const resposta = await fetch(url);

    // await 2: espera o body ser convertido em JSON
    const marcas = await resposta.json();

    const elementoLista =
        document.getElementById("lista");

    elementoLista.innerHTML = "";

    marcas.forEach(marca => {
        elementoLista.innerHTML +=
            `<li class="list-group-item">${marca.nome}</li>`;
    });
}
