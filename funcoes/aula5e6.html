// aula 05 — objetos

objeto literal

var aluno = { ... }

Um objeto agrupa propriedades relacionadas numa só variável. Cada par chave: valor é uma propriedade.

var aluno = {
  nome:         "José Silva",
  idade:        18,
  nota:         8.5,
  matriculado: true,
}
nome:"José Silva"
idade:18
nota:8.5
matriculado:true
aguardando...
objeto com arrays

let esportes = { tipo: [], jogadores: [] }

Propriedades podem ser arrays. O objeto menu também guarda listas de itens de navegação.

let esportes = {
  tipo:     [],
  jogadores: [],
}

const menu = {
  inicial: ["Home", "Login", "Contato", "Hamburger"],
  login:   ["Home", "Contato", "Hamburger"]
}
aguardando...

// spread operator

spread operator ...

esportes.tipo = [...esportes.tipo, 'Vôlei']

O ... "espalha" os itens existentes do array dentro de um novo. Assim você adiciona sem perder o que já tinha.

esportes.tipo = ['Basquete'];

esportes.tipo = [...esportes.tipo, 'Vôlei'];

console.log(esportes)
[ ] vazio — clique pra adicionar
acessar propriedade

console.log(aluno.nome)

Use a notação de ponto objeto.propriedade pra acessar qualquer valor dentro do objeto.

console.log(aluno.nome);

aluno.nome = "Dagoberto Silva";
console.log(aluno.nome);

// aula 06 — for

for · incrementar

for (let i = 0; i <= 10; i++)

Começa em 0, vai até 10 somando 1 a cada volta. O i++ é o incremento.

for (let i = 0; i <= 10; i++){
  document.write(i + ",");
}
for · decrementar

for (let i = 10; i >= 1; i--)

Começa em 10, vai até 1 subtraindo 1 a cada volta. O i-- é o decremento.

for (let i = 10; i >= 1; i--){
  document.write(i + ",");
}
for · percorrer array

for (let i = 0; i < array.length; i++)

Usa .length como limite pra percorrer cada posição do array pelo índice.

let esportes = ["Vôlei", "Basquete", "Futebol"];

for(let i = 0; i < esportes.length; i++){
  document.write(
    `Esporte: ${esportes[i]}`
  )
}

// forEach

forEach · método do array

cidades.forEach( (cidade, i, array) => { ... } )

O forEach é um método do próprio array — mais limpo que o for clássico. Recebe uma função de callback com três parâmetros: o item atual, o índice e o array inteiro.

let cidades = ["Guararapes", "Araçatuba", "Andradina"];

cidades.forEach( (cidade, i, array) => {
  document.write(
    `Cidade: ${cidade} , índice ${i} Array: ${array}`
  )
});

// while

while

while (contador < carros.length)

Verifica a condição antes de executar. Se já começa falsa, não executa nenhuma vez.

let carros = ["BMW", "Bentley", "Jaguar", "Ferrari"];
let contador = 0;

while (contador < carros.length){
  document.write(contador + carros[contador]);
  contador++;
}
do while

do { ... } while (contador2 < carros.length)

Executa o bloco pelo menos uma vez, depois verifica a condição. Diferença crucial em relação ao while.

let contador2 = 0;

do {
  document.write(carros[contador2]);
  contador2++;
} while(contador2 < carros.length);
