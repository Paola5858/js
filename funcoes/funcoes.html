  } else {
    console.log("Boa noite, Paola!");

new Date() cria um objeto com a data e hora atual do sistema. getHours() retorna a hora em formato 24h (0–23). O if / else if / else verifica a faixa e exibe a saudação certa para Paola.

let agora = new Date();
let hora = agora.getHours();
let dia  = agora.getDay();

function verificarLojaAberta() {
  if (dia === 0) {
    console.log("Fechado aos domingos.");
  } else if (hora >= 9 && hora < 18) {
    console.log("Loja aberta! 🟢");
  } else {
    console.log("Loja fechada. 🔴");
  }
}

getDay() retorna o dia da semana (0 = domingo, 6 = sábado). Combinado com getHours(), dá pra simular o horário de funcionamento real de uma loja.

// 06 — math.random · math.floor

math.random

let dado = Math.floor(Math.random())

// gera um número aleatório de 1 a 6
function rolarDado() {
  let dado = Math.floor(Math.random() * 6) + 1;
  console.log("Você tirou: " + dado);
}

Math.random() gera um decimal entre 0 e 1. Multiplicando por 6 e aplicando Math.floor() arredonda pra baixo, resultando em inteiros de 0 a 5. O + 1 ajusta o intervalo para 1–6, como um dado real.

// 07 — girar dado com faces

function girarDado() {
  let faces = ["⚀", "⚁", "⚂", "⚃", "⚄", "⚅"];
  let numeroSorteado = Math.floor(Math.random() * 6);
  return faces[numeroSorteado];
}

Math.floor(Math.random() * 6) gera um índice de 0 a 5. Esse índice acessa o array faces[], retornando o emoji correspondente — como virar um dado de verdade.

// 09 — formatar perfil

string methods

function formatarPerfil(nomeUsuario)

function formatarPerfil(nomeUsuario) {
  let nomeLimpo   = nomeUsuario.trim();
  let totalLetras = nomeLimpo.length;
  let grito       = nomeLimpo.toUpperCase();
  let sussurro    = nomeLimpo.toLowerCase();

  return "Nome: " + nomeLimpo
    + " | Letras: " + totalLetras
    + " | Grito: " + grito
    + " | Sussurro: " + sussurro;
}

trim() remove espaços extras nas bordas. length conta os caracteres. toUpperCase() e toLowerCase() alteram a capitalização — métodos essenciais para tratar e exibir dados de usuário.
